function solution(arr){
  let count = 0;
  
  for(let i = 0; i < arr.length; i++){
    for(let j = 0; j < arr.length; j++){
      const current = arr[i][j]
      if(i > 0){
        const t = arr[i - 1][j]
        if(t > current) continue
      }
      if(j < arr.length - 1){
        const r = arr[i][j + 1]
        if(r > current) continue
      }
      if(i < arr.length - 1){
        const b = arr[i + 1][j]
        if(b > current) continue
      }
      if(j > 0){
        const l = arr[i][j - 1]
        if(l > current) continue
      }
      count += 1
    }
  }
  
  return count
}

    }
    
    pointer--
    countPointer++

function solution(n, arr) {
  let answer;
  let max = Number.MIN_SAFE_INTEGER;

  for (const x of arr) {
    let sum = 0;
    let tmp = x;

    while(tmp){
      sum += tmp % 10;
      tmp = Math.floor(tmp / 10);
    }

    if (sum > max) {
      max = sum;
      answer = x;
    }

    if (sum === max) {
      if (x > answer) {
        answer = x;
      }
    }
  }

  return answer;
}

  let result = 0;

    result = result * 10 + (numCopy % 10)
